34// Helper function to print chunks
35template <typename T>
36void print_chunks(const std::vector<std::vector<T>>& chunks) {
37 std::cout << "Chunks: [" << std::endl;
38 for (size_t i = 0; i < chunks.size(); ++i) {
39 std::cout << " " << i << ": [";
40 for (const auto& value : chunks[i]) {
41 std::cout << std::fixed << std::setprecision(2) << value << " ";
42 }
43 std::cout << "]" << std::endl;
44 }
45 std::cout << "]" << std::endl;
46}
