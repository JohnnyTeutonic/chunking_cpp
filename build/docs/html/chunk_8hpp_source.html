109 // Chunk with overlap - each chunk shares n elements with the next chunk
110 std::vector<std::vector<T>> get_overlapping_chunks(size_t overlap) const {
111 if (overlap >= chunk_size_) {
112 throw std::invalid_argument("Overlap must be less than chunk size");
113 }
114
115 std::vector<std::vector<T>> chunks;
116 size_t step = chunk_size_ - overlap;
117
118 for (size_t i = 0; i + chunk_size_ <= data_.size(); i += step) {
119 chunks.push_back(std::vector<T>(data_.begin() + i, data_.begin() + i + chunk_size_));
120 }
121
122 // Handle remaining elements if any
123 if (data_.size() % step != 0) {
124 size_t remaining = data_.size() % step;
125 if (remaining > overlap) {
126 chunks.push_back(std::vector<T>(data_.end() - chunk_size_, data_.end()));
127 }
128 }
129
130 return chunks;

136 std::vector<std::vector<T>> chunk_by_predicate(Pred predicate) const {
137 std::vector<std::vector<T>> chunks;

139 return chunks;
140
141 std::vector<T> current_chunk;
142 current_chunk.push_back(data_[0]); // Add first element

145 if (predicate(data_[i])) {

159 // Chunk by sum - create chunks that sum up to a target value
160 // Note: This method ignores the original chunk_size_
161 std::vector<std::vector<T>> chunk_by_sum(T target_sum) const {
162 std::vector<std::vector<T>> chunks;
163 if (data_.empty())
164 return chunks;
165
166 std::vector<T> current_chunk;
167 T current_sum = T();
168
169 for (const T& item : data_) {
170 if (current_sum + item > target_sum && !current_chunk.empty()) {
171 chunks.push_back(current_chunk);
172 current_chunk.clear();
173 current_sum = T();
174 }
175 current_chunk.push_back(item);
176 current_sum += item;
177 }
178
179 if (!current_chunk.empty()) {
180 chunks.push_back(current_chunk);
181 }

186 // Chunk into n equal(ish) parts
187 // Note: This method ignores the original chunk_size_
188 std::vector<std::vector<T>> chunk_into_n(size_t n) const {
189 if (n == 0) {
190 throw std::invalid_argument("Number of chunks must be greater than 0");
191 }
192 if (data_.empty())
193 return std::vector<std::vector<T>>();
194
195 n = std::min(n, data_.size());
196 std::vector<std::vector<T>> chunks(n);
197 size_t base_size = data_.size() / n;
198 size_t remainder = data_.size() % n;
199
200 size_t current_pos = 0;
201 for (size_t i = 0; i < n; ++i) {
202 size_t current_chunk_size = base_size + (i < remainder ? 1 : 0);
203 chunks[i] = std::vector<T>(data_.begin() + current_pos,
204 data_.begin() + current_pos + current_chunk_size);
205 current_pos += current_chunk_size;
206 }
207
208 return chunks;

261 * @brief Chunk data into groups with similar values
262 * @param similarity_threshold Maximum difference between values in a chunk
263 * @return Vector of chunks with similar values
264 */
265 std::vector<std::vector<T>> chunk_by_similarity(T similarity_threshold) const {
266 std::vector<std::vector<T>> chunks;
267 if (data_.empty())
268 return chunks;
269
270 std::vector<T> current_chunk{data_[0]};
271 T chunk_mean = data_[0];
272
273 for (size_t i = 1; i < data_.size(); ++i) {
274 if (std::abs(data_[i] - chunk_mean) > similarity_threshold) {
275 chunks.push_back(current_chunk);
276 current_chunk.clear();
277 chunk_mean = data_[i];
278 }
279 current_chunk.push_back(data_[i]);
280 chunk_mean = std::accumulate(current_chunk.begin(), current_chunk.end(), T(0)) /
281 static_cast<T>(current_chunk.size());
282 }
283
284 if (!current_chunk.empty()) {
285 chunks.push_back(current_chunk);

293 * @return Vector of chunks where each chunk is monotonic
294 */
295 std::vector<std::vector<T>> chunk_by_monotonicity() const {
296 std::vector<std::vector<T>> chunks;
297 if (data_.size() < 2)
298 return {data_};
299
300 std::vector<T> current_chunk{data_[0]};
301 bool increasing = data_[1] > data_[0];
302
303 for (size_t i = 1; i < data_.size(); ++i) {
304 if ((data_[i] > data_[i - 1]) != increasing) {
305 chunks.push_back(current_chunk);
306 current_chunk.clear();
307 if (i < data_.size() - 1) {
308 increasing = data_[i + 1] > data_[i];
309 }
310 }
311 current_chunk.push_back(data_[i]);
312 }
313
314 if (!current_chunk.empty()) {
315 chunks.push_back(current_chunk);
316 }
317
318 return chunks;
319 }
