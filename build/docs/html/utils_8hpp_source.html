25 static T median(std::vector<T> data) {
26 if (data.empty())
27 return T();
28 std::sort(data.begin(), data.end());
29 if (data.size() % 2 == 0) {
30 // For even number of elements, return average of two middle values
31 size_t mid = data.size() / 2;
32 return (data[mid - 1] + data[mid]) / static_cast<T>(2);
33 } else {
34 // For odd number of elements, return middle value
35 return data[data.size() / 2];
36 }
37 }
38
39 static std::pair<T, size_t> mode(const std::vector<T>& data) {
40 if (data.empty())
41 return {T(), 0};
42 std::map<T, size_t> freq;
43 for (const auto& val : data)
44 freq[val]++;
45 return *std::max_element(freq.begin(), freq.end(), [](const auto& p1, const auto& p2) {
46 return p1.second < p2.second;
47 });
48 }
