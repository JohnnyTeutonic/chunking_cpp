150 std::vector<std::vector<T>> chunk_into_n(size_t n) const {
151 if (n == 0)
152 throw std::invalid_argument("Number of chunks must be greater than 0");
153 if (data.empty())
154 return std::vector<std::vector<T>>();
155 n = std::min(n, data.size());
156 std::vector<std::vector<T>> result(n);
157 size_t base_size = data.size() / n;
158 size_t remainder = data.size() % n;
159 size_t pos = 0;
160 for (size_t i = 0; i < n; ++i) {
161 size_t chunk_size = base_size + (i < remainder ? 1 : 0);
162 result[i] = std::vector<T>(data.begin() + pos, data.begin() + pos + chunk_size);
163 pos += chunk_size;
164 }
165 return result;

206 std::vector<std::vector<T>> chunk_by_similarity(T threshold) const {
207 std::vector<std::vector<T>> result;

209 return result;

211 T chunk_mean = data[0];

214 result.push_back(current_chunk);

216 chunk_mean = data[i];
217 }
218 current_chunk.push_back(data[i]);
219 chunk_mean = std::accumulate(current_chunk.begin(), current_chunk.end(), T(0)) /
