36/**
37 * @brief Strategy for chunking based on quantile thresholds
38 */
39template <typename T>
40class QuantileStrategy : public ChunkStrategy<T> {
41 double quantile_;
42
43public:
44 explicit QuantileStrategy(double quantile) : quantile_(quantile) {
45 if (quantile < 0.0 || quantile > 1.0) {
46 throw std::invalid_argument("Quantile must be between 0 and 1");
47 }
48 }
49
50 std::vector<std::vector<T>> apply(const std::vector<T>& data) const override {
51 if (data.empty())
52 return {};
53
54 // Calculate threshold value at the given quantile
55 std::vector<T> sorted_data = data;
56 std::sort(sorted_data.begin(), sorted_data.end());
57 T threshold = sorted_data[static_cast<size_t>(quantile_ * (data.size() - 1))];
58
59 // Split into exactly two chunks: values <= threshold and values > threshold
60 std::vector<T> lower_chunk;
61 std::vector<T> upper_chunk;
62
63 for (const T& value : data) {
64 if (value <= threshold) {
65 lower_chunk.push_back(value);
66 } else {
67 upper_chunk.push_back(value);
68 }
69 }
70
71 std::vector<std::vector<T>> chunks;
72 if (!lower_chunk.empty()) {
73 chunks.push_back(lower_chunk);
74 }
75 if (!upper_chunk.empty()) {
76 chunks.push_back(upper_chunk);
77 }
78
79 return chunks;
80 }
81};

84 * @brief Strategy for chunking based on variance thresholds
85 */
86template <typename T>
87class VarianceStrategy : public ChunkStrategy<T> {
88 double threshold_;
89
90 double calculate_variance(const std::vector<T>& chunk) const {
91 if (chunk.size() < 2)
92 return 0.0;
93
94 double mean = 0.0;
95 double M2 = 0.0;
96 size_t count = 0;
97
98 for (const T& value : chunk) {
99 count++;
100 double delta = value - mean;
101 mean += delta / count;
102 double delta2 = value - mean;
103 M2 += delta * delta2;
104 }
105
106 return M2 / count;
107}
108public:
109 explicit VarianceStrategy(double threshold) : threshold_(threshold) {}
110
111 std::vector<std::vector<T>> apply(const std::vector<T>& data) const override {
112 std::vector<std::vector<T>> chunks;
113 std::vector<T> current_chunk;
114
115 for (const auto& value : data) {
116 current_chunk.push_back(value);
117 if (calculate_variance(current_chunk) > threshold_) {
118 if (current_chunk.size() > 1) {
119 current_chunk.pop_back();
120 chunks.push_back(current_chunk);
121 current_chunk = {value};
122 }
123 }
124 }
125
126 if (!current_chunk.empty()) {
127 chunks.push_back(current_chunk);
128 }
129
130 return chunks;
131 }

134 double get_threshold() const {
135 return threshold_;
136 }

139 void set_threshold(double threshold) {
140 threshold_ = threshold;
141 }

145 * @brief Strategy for chunking based on entropy thresholds
146 */
147template <typename T>
148class EntropyStrategy : public ChunkStrategy<T> {
149 double threshold_;
150
151 double calculate_entropy(const std::vector<T>& chunk) const {
152 if (chunk.empty())
153 return 0.0;
154
155 std::map<T, int> freq;
156 for (const auto& val : chunk) {
157 freq[val]++;
158 }
159
160 double entropy = 0.0;
161 for (const auto& [_, count] : freq) {
162 double p = static_cast<double>(count) / chunk.size();
163 entropy -= p * std::log2(p);
164 }
165
166 return entropy;
167 }
168
169public:

195template <typename T>
196class PatternBasedStrategy : public ChunkStrategy<T> {
197private:
198 template <typename U>
199 static constexpr bool is_multidimensional_v = chunk_processing::is_multidimensional_v<U>;
200
201 template <typename U>
202 double compute_array_sum(const U& arr) const {
203 if constexpr (chunk_processing::is_multidimensional_v<U>) {
204 double sum = 0.0;
205 size_t depth = 0;
206 constexpr size_t MAX_DEPTH = 1000; // Prevent stack overflow
207
208 for (const auto& inner : arr) {
209 if (depth++ > MAX_DEPTH) {
210 throw std::runtime_error("Maximum recursion depth exceeded");
211 }
212 sum += compute_array_sum(inner);
213 }
214 return sum;
215 } else if constexpr (chunk_processing::is_vector<U>::value) {
216 return std::accumulate(arr.begin(), arr.end(), 0.0);
217 } else {
218 return static_cast<double>(arr);
219 }
220 }
221
222public:
223 // Constructor for size-based patterns
224 explicit PatternBasedStrategy(size_t pattern_size)
225 : pattern_size_(pattern_size), predicate_(nullptr) {}
226
227 // Constructor for predicate-based patterns
228 explicit PatternBasedStrategy(std::function<bool(T)> predicate)
229 : pattern_size_(0), predicate_(predicate) {}
230
231 std::vector<std::vector<T>> apply(const std::vector<T>& data) const override {
232 if (data.empty()) {
233 return {};
234 }
235
236 if constexpr (is_multidimensional_v<T>) {
237 // Handle multi-dimensional arrays
238 if (predicate_) {
239 return chunk_by_predicate(data);
240 } else {
241 return chunk_by_size(data);
242 }
243 } else {
244 // Existing single-dimension logic
245 // ...
246 }
247 }
248
249private:
250 std::vector<std::vector<T>> chunk_by_predicate(const std::vector<T>& data) const {
251 std::vector<std::vector<T>> result;
252 std::vector<T> current_chunk;
253
254 for (const auto& value : data) {
255 if constexpr (is_multidimensional_v<T>) {
256 if (predicate_(compute_array_sum(value))) {
257 if (!current_chunk.empty()) {
258 result.push_back(current_chunk);
259 current_chunk.clear();
260 }
261 }
262 } else {
263 if (predicate_(value)) {
264 if (!current_chunk.empty()) {
265 result.push_back(current_chunk);
266 current_chunk.clear();
267 }
268 }
269 }
270 current_chunk.push_back(value);
271 }
272
273 if (!current_chunk.empty()) {
274 result.push_back(current_chunk);
275 }
276
277 return result;
278 }
279
280 std::vector<std::vector<T>> chunk_by_size(const std::vector<T>& data) const {
281 if (pattern_size_ == 0) {
282 return {data};
283 }
284
285 if (data.size() < pattern_size_) {
286 return data.empty() ? std::vector<std::vector<T>>{} : std::vector<std::vector<T>>{data};
287 }
288
289 std::vector<std::vector<T>> result;
290 for (size_t i = 0; i < data.size(); i += pattern_size_) {
291 size_t end = std::min(i + pattern_size_, data.size());
292 result.emplace_back(data.begin() + i, data.begin() + end);
293 }
294 return result;
295 }
296

298 size_t get_pattern_size() const {
299 return pattern_size_;
300 }

302 return predicate_;
303 }

307 pattern_size_ = size;
308 }

310 predicate_ = std::move(pred);
311 }

313private:
314 size_t pattern_size_;

354public:

356 : similarity_threshold_(similarity_threshold), size_threshold_(size_threshold) {}
357
358 std::vector<std::vector<T>> apply(const std::vector<T>& data) const override {
359 // Step 1: Similarity-based chunking
360 chunk_processing::Chunk<T> initial_chunker(1);
361 initial_chunker.add(data);
362 auto similarity_chunks = initial_chunker.chunk_by_threshold(similarity_threshold_);
363
364 // Step 2: Size-based chunking
365 std::vector<std::vector<T>> result;
366 for (const auto& chunk : similarity_chunks) {
367 chunk_processing::Chunk<T> size_chunker(size_threshold_);
368 size_chunker.add(chunk);
369 auto sized_chunks = size_chunker.get_chunks();
370 result.insert(result.end(), sized_chunks.begin(), sized_chunks.end());
371 }
372 return result;
373 }
