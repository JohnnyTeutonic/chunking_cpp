44// Helper function to print chunks
45template <typename T>
46void print_chunks(const std::vector<std::vector<T>>& chunks) {
47 std::cout << "Chunks: [" << std::endl;
48 for (size_t i = 0; i < chunks.size(); ++i) {
49 std::cout << " " << i << ": [";
50 for (const auto& value : chunks[i]) {
51 std::cout << std::fixed << std::setprecision(2) << value << " ";
52 }
53 std::cout << "]" << std::endl;
54 }
55 std::cout << "]" << std::endl;
56}

58/**
59 * @brief Helper function to print sub-chunks with detailed formatting
60 * @tparam T The type of elements in the chunks
61 * @param sub_chunks The 3D vector containing the sub-chunks to print
62 * @param label The label to display for this set of sub-chunks
63 * @param precision The number of decimal places to show for floating-point numbers
64 */
65template <typename T>
66void print_sub_chunks(const std::vector<std::vector<std::vector<T>>>& sub_chunks,
67 const std::string& label, int precision = 2) {
68 std::cout << "\n" << label << ":\n";
69 for (size_t i = 0; i < sub_chunks.size(); ++i) {
70 std::cout << "Level " << i + 1 << ":\n";
71 for (size_t j = 0; j < sub_chunks[i].size(); ++j) {
72 std::cout << " Sub-chunk " << j + 1 << ": ";
73 for (const auto& val : sub_chunks[i][j]) {
74 std::cout << std::fixed << std::setprecision(precision) << val << " ";
75 }
76 std::cout << "\n";
77 }
78 }
79}

132/**
133 * @brief Demonstrates adaptive conditional sub-chunking
134 *
135 * This function shows how to use conditional sub-chunking with
136 * adaptive thresholds based on chunk properties.
137 */
138void demonstrate_adaptive_conditional_subchunking() {
139 std::cout << "\n=== Adaptive Conditional Sub-chunking ===" << std::endl;
140
141 // Example data
142 std::vector<std::vector<double>> data = {
143 {1.0, 1.1, 1.2, 5.0, 5.1, 5.2},
144 {2.0, 2.1, 2.2, 10.0, 10.1, 10.2},
145 {3.0, 3.1, 3.2, 15.0, 15.1, 15.2}
146 };
147
148 // Conditional sub-chunking
149 auto condition = [](const std::vector<double>& chunk) {
150 double mean = std::accumulate(chunk.begin(), chunk.end(), 0.0) / chunk.size();
151 double variance = 0.0;
152 for (const auto& val : chunk) {
153 variance += (val - mean) * (val - mean);
154 }
155 variance /= chunk.size();
156 return variance > 50.0;
157 };
158
159 auto variance_strategy = std::make_shared<VarianceStrategy<double>>(5.0);
160 ConditionalSubChunkStrategy<double> conditional_strategy(variance_strategy, condition, 2);
161 auto conditional_result = conditional_strategy.apply(data);
162
163 print_sub_chunks(conditional_result, "Conditional Sub-chunking");
164}
