48 * @brief Strategy for applying recursive sub-chunking
49 * @tparam T The type of elements to process
50 */
51template <typename T>
52class RecursiveSubChunkStrategy : public SubChunkStrategy<T> {
53private:
54 std::shared_ptr<ChunkStrategy<T>> strategy_;
55 size_t max_depth_;
56 size_t min_chunk_size_;
57
58public:
59 /**
60 * @brief Construct a recursive sub-chunking strategy
61 * @param strategy The base chunking strategy to apply recursively
62 * @param max_depth Maximum recursion depth
63 * @param min_chunk_size Minimum size for chunks to be processed
64 */
65 RecursiveSubChunkStrategy(std::shared_ptr<ChunkStrategy<T>> strategy, size_t max_depth = 2,
66 size_t min_chunk_size = 1)
67 : strategy_(strategy), max_depth_(max_depth), min_chunk_size_(min_chunk_size) {}
68
69 /**
70 * @brief Apply recursive sub-chunking to the input chunks
71 * @param chunks The input chunks to process
72 * @return A vector of vectors of vectors containing the recursively sub-chunked data
73 */
74 std::vector<std::vector<std::vector<T>>>
75 apply(const std::vector<std::vector<T>>& chunks) override {
76 return apply_recursive(chunks, 1);
77 }
78
79private:
80 /**
81 * @brief Helper function for recursive chunking
82 * @param chunks The chunks to process
83 * @param current_depth Current recursion depth
84 * @return Processed sub-chunks
85 */
86 std::vector<std::vector<std::vector<T>>>
87 apply_recursive(const std::vector<std::vector<T>>& chunks, size_t current_depth) {
88 /**
89 * @details This method recursively applies the chunking strategy:
90 * 1. Checks depth and size constraints
91 * 2. Applies base strategy to each chunk
92 * 3. Recursively processes sub-chunks if needed
93 * 4. Returns the hierarchical result
94 */
95 std::vector<std::vector<std::vector<T>>> result;
96
97 for (const auto& chunk : chunks) {
98 if (current_depth >= max_depth_ || chunk.size() <= min_chunk_size_) {
99 result.push_back({chunk});
100 continue;
101 }
102
103 auto sub_chunks = strategy_->apply(chunk);
104 if (current_depth < max_depth_) {
105 auto recursive_result = apply_recursive(sub_chunks, current_depth + 1);
106 result.insert(result.end(), recursive_result.begin(), recursive_result.end());
107 } else {
108 result.push_back(sub_chunks);
109 }
110 }
111
112 return result;
113 }
114};

136 /**
137 * @brief Apply hierarchical sub-chunking to the input chunks
138 * @param chunks The input chunks to process
139 * @return A vector of vectors of vectors containing the hierarchically sub-chunked data
140 */
141 std::vector<std::vector<std::vector<T>>>
142 apply(const std::vector<std::vector<T>>& chunks) override {
143 return apply_hierarchical(chunks, 0);
144 }
145
146private:
147 /**
148 * @brief Helper function for hierarchical chunking
149 * @param chunks The chunks to process
150 * @param level Current hierarchy level
151 * @return Processed sub-chunks
152 */
153 std::vector<std::vector<std::vector<T>>>
154 apply_hierarchical(const std::vector<std::vector<T>>& chunks, size_t level) {
155 std::vector<std::vector<std::vector<T>>> result;
156
157 if (level >= strategies_.size()) {
158 result.push_back(chunks);
159 return result;
160 }
161
162 for (const auto& chunk : chunks) {
163 if (chunk.size() <= min_chunk_size_) {
164 result.push_back({chunk});
165 continue;
166 }
167
168 auto sub_chunks = strategies_[level]->apply(chunk);
169 auto next_level_chunks = apply_hierarchical(sub_chunks, level + 1);
170 result.insert(result.end(), next_level_chunks.begin(), next_level_chunks.end());
171 }
172
173 return result;
174 }

195 ConditionalSubChunkStrategy(std::shared_ptr<ChunkStrategy<T>> strategy,

197 size_t min_chunk_size = 1)
198 : strategy_(strategy), condition_(condition), min_chunk_size_(min_chunk_size) {}
199
200 /**
201 * @brief Apply conditional sub-chunking to the input chunks
202 * @param chunks The input chunks to process
203 * @return A vector of vectors of vectors containing the conditionally sub-chunked data
204 */
205 std::vector<std::vector<std::vector<T>>>
206 apply(const std::vector<std::vector<T>>& chunks) override {
207 std::vector<std::vector<std::vector<T>>> result;
208
209 for (const auto& chunk : chunks) {
210 if (chunk.size() <= min_chunk_size_ || !condition_(chunk)) {
211 result.push_back({chunk});
212 continue;
213 }
214
215 auto sub_chunks = strategy_->apply(chunk);
216 result.push_back(sub_chunks);
217 }
218
219 return result;
